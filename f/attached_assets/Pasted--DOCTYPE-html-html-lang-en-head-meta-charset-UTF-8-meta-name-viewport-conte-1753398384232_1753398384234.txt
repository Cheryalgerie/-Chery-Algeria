    // وظيفة البحث في المنتجات
    function searchProducts() {
      const searchTerm = document.getElementById('search-input').value.toLowerCase();
      const products = document.querySelectorAll('.product');
      const loading = document.getElementById('loading');
      
      if (searchTerm.length === 0) {
        products.forEach(product => {
          product.style.display = 'block';
        });
        return;
      }
      
      // عرض تأثير التحميل
      loading.style.display = 'block';
      
      setTimeout(() => {
        let hasResults = false;
        
        products.forEach(product => {
          const productName = product.dataset.name.toLowerCase();
          if (productName.includes(searchTerm)) {
            product.style.display = 'block';
            product.style.animation = 'fadeIn 0.5s ease-in';
            hasResults = true;
          } else {
            product.style.display = 'none';
          }
        });
        
        loading.style.display = 'none';
        
        if (!hasResults) {
          showNoResults();
        } else {
          hideNoResults();
        }
      }, 800);
    }

    // وظيفة المفضلة
    let wishlist = [];
    
    function toggleWishlist(button) {
      const product = button.closest('.product');
      const productName = product.dataset.name;
      
      if (button.classList.contains('active')) {
        // إزالة من المفضلة
        button.classList.remove('active');
        button.innerHTML = '♡';
        wishlist = wishlist.filter(item => item !== productName);
        showNotification('Product removed from wishlist', 'info');
      } else {
        // إضافة للمفضلة
        button.classList.add('active');
        button.innerHTML = '♥';
        wishlist.push(productName);
        showNotification('Product added to wishlist', 'success');
      }
      
      // حفظ في التخزين المحلي
      localStorage.setItem('sabwear-wishlist', JSON.stringify(wishlist));
    }
    
    function loadWishlist() {
      const saved = localStorage.getItem('sabwear-wishlist');
      if (saved) {
        wishlist = JSON.parse(saved);
        wishlist.forEach(productName => {
          const product = document.querySelector(`[data-name="${productName}"]`);
          if (product) {
            const button = product.querySelector('.wishlist-btn');
            button.classList.add('active');
            button.innerHTML = '♥';
          }
        });
      }
    }
